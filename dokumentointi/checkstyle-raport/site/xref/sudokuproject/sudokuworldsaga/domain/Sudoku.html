32      /**
33       * Make new sudoku based on values listed on parameter sudokuDataArray
34       * 
35       * @param cols Sudoku columns (>1)
36       * @param rows Sudoku rows (>1)
37       * @param sudokuDataArray Sudoku data
38       *
39       * @throws IllegalArgumentException When invalid sudoku dimensions or sudokuData given.
40      */
41      
42      public Sudoku(int cols, int rows, int[][] sudokuDataArray) throws IllegalArgumentException {
43          if (cols <= 1 || rows <= 1) {
44              throw (new IllegalArgumentException("Too small"));
45          }
46          if (sudokuDataArray == null) {
47              throw (new IllegalArgumentException("Datastring null"));
48          }
49          this.rows = rows;
50          this.cols = cols;
51          sudoku = new SudokuData(cols, rows, sudokuDataArray);
52      }

55       * Make new empty sudoku with all cell values set to zero

57       * @param cols Sudoku columns (>1)
58       * @param rows Sudoku rows (>1)

60       * @throws IllegalArgumentException When invalid sudoku dimensions given.
61      */
62      
63      public Sudoku(int cols, int rows) throws IllegalArgumentException {
64          if (cols <= 1 || rows <= 1) {
65              throw (new IllegalArgumentException("Too small"));
66          }
67          this.rows = rows;
68          this.cols = cols;
69          this.sudoku = new SudokuData(cols, rows);
70      }

72      /**
73       * Clones a sudoku from the values of the sudoku given as parameter
74      */
75      public Sudoku(Sudoku sudoku) throws IllegalArgumentException {
76          if (sudoku == null) {
77              throw new IllegalArgumentException("Source sudoku is null");
78          }
79          this.cols = sudoku.cols;
80          this.rows = sudoku.rows;
81          this.sudoku = new SudokuData(cols, rows, sudoku.getSudokuData());

83      /**
84       * Method goes through whole sudoku and checks if there are any unsolved sudoku cells.
85       * 
86       * @return true if solved, false if not solved
87       */
88      public boolean isSolved() {
89          return countUnsolved() == 0;
90      }
91      
92      /**
93       * Method goes through whole sudoku and counts the unsolved sudoku cells.
94       * 
95       * @return number of unsolved sudoku cells
96       */
97      public int countUnsolved() {
98          int n = 0;
99          int sudokuSize = cols * rows;
100         for (int i = 0; i < sudokuSize; i++) {
101             for (int j = 0; j < sudokuSize; j++) {
102                 if (sudoku.getXY(i, j) == 0) {
103                     n++;
104                 }
105             }
106         }
107         return n;
108     }

127     /**
128      * Takes sudoku cell location (x, y) as parameteres and finds all the possible entry values to that cell.
129      * 
130      * @param x column
131      * @param y row
132      * @return Returns an ArrayList of possible entries in the cell (x, y)
133      * @throws IllegalArgumentException if x or y out of bounds
134      */
135     
136     public ArrayList<Integer> getEntries(int x, int y) throws IllegalArgumentException {
137         if (!isInRange(x, y)) {
138             throw new IllegalArgumentException("x and/or y not in range");
139         }
140         ArrayList<Integer> validEntries = new ArrayList<Integer>();
141         ArrayList<Integer> falseEntries = new ArrayList<Integer>();
142         // find all the numbers already in the subset
143         
144         // FIXED VALUE 10 SHOULD BE DYNAMIC DEPENDING ON SUDOKU SIZE
145         // [1,...,9,a,...z] (scandics?) 
146         
147         for (int i = 1; i < 10; i++) {
148             validEntries.add(i);
149         }
150         for (int i : sudoku.getNumbersInSubset(x, y)) {
151             if (!falseEntries.contains(i) && i != 0) {
152                 falseEntries.add(i);
153             }
154         }
155         // find all the numbers already on the row
156         for (int i = 0; i < rows * cols; i++) {
157             if (!falseEntries.contains(getXY(i, y)) && getXY(i, y) != 0) {
158                 falseEntries.add(getXY(i, y));
159             }
160         }
161         // find all the numbers already on the column
162         for (int i = 0; i < rows * cols; i++) {
163             if (!falseEntries.contains(getXY(x, i)) && getXY(x, i) != 0) {
164                 falseEntries.add(getXY(x, i));
165             }
166         }
167         validEntries.removeAll(falseEntries);
168         return validEntries;
169     }

227     @Override
228     public String toString() {
229         StringBuilder sudokuString = new StringBuilder("");
230         String column = "|";
231         for (int i = 0; i < rows * cols; i++) {
232             if (i % rows == 0 && i != 0) {
233                 sudokuString.append(lineRow());
234             }
235             for (int j = 0; j < rows * cols; j++) {
236                 if (j % cols == 0 && j != 0) {
237                     sudokuString.append(column);
238                 }
239                 sudokuString.append(sudoku.getSymbolXY(j, i));
240             }
241             sudokuString.append("\n");
242         }
243         sudokuString.append("Unsolved: " + countUnsolved() + "\n");
244         return sudokuString.toString();
245     }

254         }

257             return false;
258         }

264     /**
265      * Returns true/false if (x,y) in sudoku range 
266      */
267 
268     private boolean isInRange(int x, int y) {
269         int size = this.getSize();
270         return (((y < 0 || y > size) || (x < 0 || x > size)) ? false : true);
271     }
272     
273     /**
274      * Prints a row of '-' for toString() method
275      */
276     
277     private String lineRow() {
278         StringBuilder string = new StringBuilder("");
279         String c = "-";
280         for (int i = 0; i < (cols + 1) * rows - 1; i++) {
281             string.append(c);
282         }
283         string.append("\n");
284         return string.toString();
285     }
