338     private void solveButtonActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_solveButtonActionPerformed
339         // Action when solve Sudoku button is clicked
340         Sudoku solvedSudoku = SudokuSolver.solve(sudoku);

342             sudokuPanel1.updateSolvedValues(sudoku, solvedSudoku);

345         } else {
346             textField.append("Sudoku is not solvable. Mistakes were made.\n");

359     private void loadSudokuButtonActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_loadSudokuButtonActionPerformed
360         // Action when load Sudoku button is clicked
361         // sudokuArea.setText("Opening");

388     private void difficultyValueTextKeyPressed(java.awt.event.KeyEvent evt) {//GEN-FIRST:event_difficultyValueTextKeyPressed
389         if (evt.getKeyCode() == java.awt.event.KeyEvent.VK_ENTER) {
390             int value = 0;
391             try {
392                 value = Integer.parseInt(difficultyValueText.getText());
393                 if (value < 0) {
394                     value = 0;
395                     difficultyValueText.setText(value + "");
396                 } else if (value > 60) {
397                     value = 60;
398                     difficultyValueText.setText(value + "");
399                 }
400             } catch (Exception ex) {
401                 value = 0;
402                 difficultyValueText.setText("0");
403             }
404             difficultySlider.setValue(value);
405             difficultyValue = value;
406         }
407     }//GEN-LAST:event_difficultyValueTextKeyPressed

409     private void sudokuPanel1MouseClicked(java.awt.event.MouseEvent evt) {//GEN-FIRST:event_sudokuPanel1MouseClicked
410         disableSudokuButtons();
411         int[] xy = coordsToXY(evt.getX(), evt.getY());  
412         if (xy[0] < 9 && xy[0] >= 0 && xy[1] < 9 && xy[1] >= 0) {
413             activeX = xy[0];
414             activeY = xy[1];
415             textField.append("Selected cell x: " + activeX + ", y: " + activeY + " with value: " + sudoku.getXY(activeX, activeY) + "\n");
416             for (int i : sudoku.getEntries(activeX, activeY)) {
417                 sudokuButtons[i].setEnabled(true);
418             }
419             if (sudoku.getXY(activeX, activeY) != 0) {
420                 sudokuButtons[0].setEnabled(true);
421             }
422         }
423     }//GEN-LAST:event_sudokuPanel1MouseClicked

425     private void sudokuButtonActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_sudokuButtonActionPerformed

441             }
442             sudoku.set(activeX, activeY, val);
443             textField.append("Changed cell x: " + activeX + ", y: " + activeY + " value to: " + sudoku.getXY(activeX, activeY) + "\n");
444         }
445         disableSudokuButtons();
446     }//GEN-LAST:event_sudokuButtonActionPerformed

448     // Calculates the sudoku row/col from mouse click coords
449     private int[] coordsToXY(int clickX, int clickY) {
450         // constants that depend on UI construction
451         int borderWidth = 5;
452         int cellWidth = 45;
453         
454         int absX = clickX - borderWidth;
455         int absY = clickY - borderWidth;
456         
457         //textField.append("location: x = " + absX + ", y = " + absY + "\n");
458         
459         // Borders on sudoku subsets have to be removed from coords
460         absX = removeBorders(absX, borderWidth, cellWidth);
461         absY = removeBorders(absY, borderWidth, cellWidth);
462         
463         // Calculate row/col index
464         int x = (absX - (absX % cellWidth)) / cellWidth;
465         int y = (absY - (absY % cellWidth)) / cellWidth;
466         
467         return new int[]{x, y};
468     }
469     
470     private int removeBorders(int coord, int borderWidth, int cellWidth) {
471         if (coord - (2 * borderWidth) >= cellWidth * 3 && coord - (2 * borderWidth) < cellWidth * 6) {
472             // if click is on cols 3-5, remove 1 extra border from x coords
473             coord -= (2 * borderWidth);
474         } else if (coord - (4 * borderWidth) > cellWidth * 6 && coord - (4 * borderWidth) < cellWidth * 9) {
475             // if click is on cols 6-8, remove 2 extra border from x coords
476             coord -= 4 * borderWidth;
477         }
478         return coord;
479     }
