17      /** 
18       * Sets the value of the cell as param newValue. If newValue == "0" then the cell value is set to "".
19       * 
20       * @param newValue The value to be inserted in the cell.
21       */
22      public void setValue(String newValue) {
23          if (newValue == null) {
24              return;
25          }
26          try {
27              int val = Integer.parseInt(newValue);
28              if (val < 0 || val > 9) {
29                  return;
30              }
31          } catch (NumberFormatException ex) {
32              return;
33          }
34          if (newValue.equals("0")) {
35              cellValue.setText("");
36          } else {
37              cellValue.setText(newValue);
38          }
39      }
