22      /**
23       * updateValues() updates the SudokuPanel with values from sudoku
24       * 
25       * @param sudoku 
26       */
27      public void updateValues(Sudoku sudoku) {
28          if (sudoku == null) {
29              return;
30          }
31          for (int i = 0; i < 9; i++) {
32              for (int j = 0; j < 9; j++) {
33                  setXY(i, j, sudoku.getXY(i, j) + "");
34                  setXYColor(i, j, Color.WHITE);
35              }
36          }
37      }
38  
39      /**
40       * updateSolvedValues() updates the empty cells in SudokuPanel with the values from solvedSudoku
41       * 
42       * @param sudoku Unsolved sudoku.
43       * @param solvedSudoku Fully solved sudoku.
44       * 
45       */
46      public void updateSolvedValues(Sudoku sudoku, Sudoku solvedSudoku) {
47          if (sudoku == null || solvedSudoku == null) {
48              return;
49          }
50          for (int i = 0; i < 9; i++) {
51              for (int j = 0; j < 9; j++) {
52                  if (sudoku.getXY(i, j) == 0) {
53                      setXY(i, j, solvedSudoku.getXY(i, j) + "");
54                      setXYColor(i, j, Color.GREEN);
55                  }
56              }
57          }
58      }

60      /**
61       * Changes the value of cell (x, y) to param value.
62       * 
63       * @param x column
64       * @param y row
65       * @param value new value
66       */
67      public void setXY(int x, int y, String value) {
68          if (notInRange(x, 0, 8)) {
69              return;
70          }
71          if (notInRange(y, 0, 8)) {
72              return;
73          }
74          subsets[findSubsetIndex(x, y)].setValue(findCellIndex(x, y), value);
75      }
76      
77      /**
78       * Changes the color of cell (x, y) to param color.
79       * 
80       * @param x column
81       * @param y row
82       * @param color new color
83       */
84      public void setXYColor(int x, int y, Color color) {
85          if (notInRange(x, 0, 8) || notInRange(y, 0, 8)) {
86              return;
87          }
88          subsets[findSubsetIndex(x, y)].setColor(findCellIndex(x, y), color);
89      }
90      
91      private int findSubsetIndex(int x, int y) {
92          int rowFactor = ((y + 1) - y % 3) / 3;
93          int columnFactor = ((x + 1) - x % 3) / 3;
94          return columnFactor + 3 * rowFactor;
95      }
96      
97      private int findCellIndex(int x, int y) {
98          int row = y % 3;
99          int col = x % 3;
100         return col + 3 * row;
101     }
102     
103     private boolean notInRange(int i, int lower, int upper) {
104         if (i < lower || i > upper) {
105             return true;
106         }
107         return false;
108     }
